## Pre-implementation Notes

- `createSupabaseAdminClient()` を `src/lib/supabase/admin.ts` からインポートして使用（D-02）
- articlesテーブルの現在スキーマ（schema.sql）には `image_url` 列が**存在しない**。このプランで追加する
- `StoredFeed` 型（`{ items: FeedItem[], imageUrl?: string }`）の `imageUrl` はチャンネルアイコン（`members.image_url`）であり、記事ごとの thumbnail（`articles.image_url`）とは別物
- `saveArticles(substackId, items, imageUrl?)` の `imageUrl` はチャンネルアイコン → `members.image_url` をUPDATEする（D-04）
- 記事の重複排除は `ON CONFLICT (link) DO NOTHING` で実装する（D-03、schema.sqlの UNIQUE(link) 制約を利用）
- Supabase JS の `upsert({ onConflict: 'link', ignoreDuplicates: true })` でこれを実現する
- `fetchFeed.ts` は `getArticles()` を呼ぶだけなので変更不要（D-05の「import変更ゼロ」を維持）

### Task 2: `src/lib/kvArticles.ts` を Supabase実装に書き換える

**File:** `src/lib/kvArticles.ts`（全書き換え）

> **注意:** `redis` インポートと `StoredFeed` 型の内部実装を全置換する。
> `StoredFeed` 型は `{ items: FeedItem[], imageUrl?: string }` のまま維持（fetchFeed.ts との互換性）。

```typescript
import { createSupabaseAdminClient } from './supabase/admin'
import type { FeedItem } from './types'

type StoredFeed = { items: FeedItem[]; imageUrl?: string }

export async function getArticles(substackId: string): Promise<StoredFeed> {
  const supabase = createSupabaseAdminClient()
  const [articlesResult, memberResult] = await Promise.all([
    supabase
      .from('articles')
      .select('title, link, pub_date, image_url')
      .eq('substack_id', substackId)
      .order('pub_date', { ascending: false }),
    supabase
      .from('members')
      .select('image_url')
      .eq('substack_id', substackId)
      .maybeSingle(),
  ])

  const items: FeedItem[] = (articlesResult.data ?? []).map((a: any) => ({
    title: a.title ?? undefined,
    link: a.link,
    isoDate: a.pub_date ?? undefined,
    thumbnail: a.image_url ?? undefined,
  }))

  return {
    items,
    imageUrl: memberResult.data?.image_url ?? undefined,
  }
}

export async function deleteArticles(substackId: string): Promise<void> {
  const supabase = createSupabaseAdminClient()
  const { error } = await supabase
    .from('articles')
    .delete()
    .eq('substack_id', substackId)
  if (error) throw error
}

// ON CONFLICT (link) DO NOTHING で重複排除（D-03, schema.sql の UNIQUE(link) 制約を利用）
// imageUrl が渡された場合は members.image_url を更新する（D-04）
export async function saveArticles(
  substackId: string,
  newItems: FeedItem[],
  imageUrl?: string
): Promise<void> {
  const supabase = createSupabaseAdminClient()

  const validItems = newItems.filter((item) => item.link)
  if (validItems.length > 0) {
    const { error } = await supabase.from('articles').upsert(
      validItems.map((item) => ({
        substack_id: substackId,
        title: item.title ?? null,
        link: item.link!,
        pub_date: item.isoDate ?? null,
        image_url: item.thumbnail ?? null,
      })),
      { onConflict: 'link', ignoreDuplicates: true }
    )
    if (error) throw error
  }

  if (imageUrl !== undefined) {
    const { error } = await supabase
      .from('members')
      .update({ image_url: imageUrl })
      .eq('substack_id', substackId)
    if (error) throw error
  }
}
```
