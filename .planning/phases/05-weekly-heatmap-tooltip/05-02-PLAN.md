```typescript
'use client'
import { useState, useRef, useEffect } from 'react'

// withUtm関数:
function withUtm(url: string): string {
  if (!url || url === '#') return url
  const sep = url.includes('?') ? '&' : '?'
  return `${url}${sep}utm_source=keep-substack&utm_medium=referral`
}

// click outside検知:
useEffect(() => {
  if (!open) return
  function handleClickOutside(e: MouseEvent) {
    if (cellRef.current && !cellRef.current.contains(e.target as Node)) {
      setOpen(false)
    }
  }
  document.addEventListener('mousedown', handleClickOutside)
  return () => document.removeEventListener('mousedown', handleClickOutside)
}, [open])

// onMouseLeaveでの範囲チェック:
onMouseLeave={(e) => {
  if (cellRef.current && !cellRef.current.contains(e.relatedTarget as Node)) {
    setOpen(false)
  }
}}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Server Component → Client Component (HeatmapTooltip) | propsとしてarticles配列（title, link, thumbnail）がブラウザに送信される |
| HeatmapTooltip → 外部記事URL | thumbnailのimg srcとして外部Substack URLが使われる |
| HeatmapTooltip → リンク遷移 | article.linkがaタグのhrefに使われる |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-04 | Tampering | HeatmapTooltip.tsx: article.link → href | mitigate | `withUtm(article.link ?? '#')` でnullガード。target="_blank" + rel="noopener noreferrer" で親ウィンドウへのアクセスを遮断（TIP-02要件） |
| T-05-05 | Information Disclosure | HeatmapTooltip.tsx: thumbnailをimg srcに使用 | accept | img srcは画像読み込みのみ。JavaScriptを実行しない。Substackは信頼できるソース。ASVS L1範囲外。 |
| T-05-06 | Elevation of Privilege | page.tsx: getMembers()の取得データをWeeklyHeatmapGridに全量渡す | accept | メンバーデータ（name, substackId, teamId, addedAt）は公開情報。認証・認可は本フェーズのスコープ外（Phase 6 ADM-03で対応）。 |
| T-05-07 | Denial of Service | WeeklyHeatmapGrid: 50人×7日のセル全域に対しHeatmapTooltipがhydrate | mitigate | HeatmapTooltipを記事ありセルのみに適用（count === 0はdivのみ）。不要なClient Componentのハイドレーションを最小化する。 |
</threat_model>
