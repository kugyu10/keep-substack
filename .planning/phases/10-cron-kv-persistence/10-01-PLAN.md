```typescript
import { redis } from './redis'
import type { FeedItem } from './types'

// D-01: KVキー = articles:{substackId}
export async function getArticles(substackId: string): Promise<FeedItem[]> {
  const articles = await redis.get<FeedItem[]>(`articles:${substackId}`)
  return articles ?? []
}

// D-05: article.link URL でdedupe。既存KVに同じlinkがあればスキップ
export async function saveArticles(
  substackId: string,
  newItems: FeedItem[]
): Promise<void> {
  const existing = await getArticles(substackId)
  const existingLinks = new Set(existing.map((a) => a.link))
  const toAdd = newItems.filter((item) => item.link && !existingLinks.has(item.link))
  if (toAdd.length === 0) return
  await redis.set(`articles:${substackId}`, [...existing, ...toAdd])
}
```

注意点:
- `link` が undefined のアイテムは dedupe 対象外（filter で除外: `item.link && !existingLinks.has(item.link)`）
- D-06 KVサイズ上限対策は実装しない（YAGNI）
- JSON.stringify/parse は不要（@upstash/redis が自動シリアライズ）

ディレクトリ src/app/api/cron/ が存在しない場合は作成してからファイルを作成する。

```typescript
import type { NextRequest } from 'next/server'
import { getMembers } from '@/lib/kvMembers'
import { fetchWithRetry } from '@/lib/fetchFeed'
import { saveArticles } from '@/lib/kvArticles'

// D-04: CRON_SECRET Bearer認証（nullチェック先行でバイパス防止）
export async function GET(request: NextRequest) {
  const authHeader = request.headers.get('authorization')
  const cronSecret = process.env.CRON_SECRET

  if (!cronSecret || authHeader !== `Bearer ${cronSecret}`) {
    return new Response('Unauthorized', { status: 401 })
  }

  const members = await getMembers()

  // Promise.allSettled: 1件の失敗で全体を止めない
  await Promise.allSettled(
    members.map(async (m) => {
      const { items } = await fetchWithRetry(
        `https://${m.substackId}.substack.com/feed`
      )
      await saveArticles(m.substackId, items)
    })
  )

  return Response.json({ ok: true, count: members.length })
}
```

注意点:
- `!cronSecret` チェックを先に行うことで CRON_SECRET 未設定時の認証バイパスを防ぐ（Pitfall 2）
- `Promise.allSettled` により1件失敗しても他のメンバーの処理を継続する（Pitfall 4）
